<objective>
Create the liveness narration engine that converts observer events into user-facing progress frames. The NarrationObserver wraps an inner Observer, translates tool_call_start/turn_stage events into semantic NarrationFrame structs, and delivers them through a callback. This is the "thinking out loud" infrastructure for Claude Code parity.

Purpose: REQ-019 requires the agent to emit real-time status about what it's doing, which tool it picked, why, and what it's waiting on. The existing VerboseObserver writes to stderr for CLI debug — NarrationObserver is the channel-aware, user-facing version that channels and SSE (Phase 2) will consume.

Output: `src/agent/narration.zig` with NarrationObserver implementing Observer vtable, narration_frame variant in ObserverEvent, and wiring in root.zig.
</objective>

    **Step 2: Create src/agent/narration.zig**

    Create the NarrationObserver that:
    1. Wraps an inner `Observer` (forwarding all events)
    2. On specific events, also emits a `narration_frame` event through the inner observer
    3. Uses a `NarrationCallback` fn pointer for direct frame delivery (optional, can be null)

    ```zig
    const std = @import("std");
    const observability = @import("../observability.zig");
    const Observer = observability.Observer;
    const ObserverEvent = observability.ObserverEvent;
    const ObserverMetric = observability.ObserverMetric;

    pub const NarrationCallback = *const fn (ctx: *anyopaque, frame: NarrationFrame) void;

    pub const NarrationFrame = struct {
        message: []const u8,
        frame_type: ObserverEvent.narration_frame.FrameType,  // reuse the enum from ObserverEvent
        tool_name: ?[]const u8 = null,
        step_index: ?u32 = null,
        step_total: ?u32 = null,
    };

    // NOTE: Use the FrameType from the ObserverEvent narration_frame struct field's type.
    // Access it as: ObserverEvent.narration_frame  — this is the struct type of that union variant.
    // The FrameType enum lives inside that struct.
    // However, if Zig's type system requires a different approach, define FrameType at module level
    // and reference it from both places.

    pub const NarrationObserver = struct {
        inner: Observer,
        callback: ?NarrationCallback = null,
        callback_ctx: ?*anyopaque = null,

        const vtable = Observer.VTable{
            .record_event = recordEvent,
            .record_metric = recordMetric,
            .flush = flush,
            .name = name,
        };

        pub fn observer(self: *NarrationObserver) Observer {
            return .{
                .ptr = @ptrCast(self),
                .vtable = &vtable,
            };
        }

        fn recordEvent(ptr: *anyopaque, event: *const ObserverEvent) void {
            const self: *NarrationObserver = @ptrCast(@alignCast(ptr));
            // Always forward to inner observer
            self.inner.recordEvent(event);

            // Generate narration frames for specific events
            switch (event.*) {
                .tool_call_start => |e| {
                    self.emitFrame(.{
                        .message = "Using tool",
                        .frame_type = .tool_start,
                        .tool_name = e.tool,
                    });
                },
                .tool_call => |e| {
                    self.emitFrame(.{
                        .message = if (e.success) "Tool completed" else "Tool failed",
                        .frame_type = if (e.success) .tool_done else .error_recovery,
                        .tool_name = e.tool,
                    });
                },
                .turn_stage => |e| {
                    const label = turnStageToNarration(e.stage);
                    if (label) |msg| {
                        const ftype = turnStageToFrameType(e.stage);
                        self.emitFrame(.{
                            .message = msg,
                            .frame_type = ftype,
                        });
                    }
                },
                else => {},
            }
        }

        fn emitFrame(self: *NarrationObserver, frame: NarrationFrame) void {
            if (self.callback) |cb| {
                cb(self.callback_ctx orelse @ptrCast(&self.inner), frame);
            }
            // Also emit as observer event for any downstream consumers
            const narration_event = ObserverEvent{ .narration_frame = .{
                .message = frame.message,
                .frame_type = frame.frame_type,
                .tool_name = frame.tool_name,
                .step_index = frame.step_index,
                .step_total = frame.step_total,
            } };
            self.inner.recordEvent(&narration_event);
        }

        fn recordMetric(ptr: *anyopaque, metric: *const ObserverMetric) void {
            const self: *NarrationObserver = @ptrCast(@alignCast(ptr));
            self.inner.recordMetric(metric);
        }

        fn flush(ptr: *anyopaque) void {
            const self: *NarrationObserver = @ptrCast(@alignCast(ptr));
            self.inner.flush();
        }

        fn name(_: *anyopaque) []const u8 {
            return "narration";
        }

        fn turnStageToNarration(stage: []const u8) ?[]const u8 {
            if (std.mem.eql(u8, stage, "turn_start")) return "Gathering context";
            if (std.mem.eql(u8, stage, "memory_enrich")) return "Retrieving memory";
            if (std.mem.eql(u8, stage, "build_provider_messages")) return "Preparing request";
            if (std.mem.eql(u8, stage, "dispatch_tools")) return "Running tools";
            if (std.mem.eql(u8, stage, "tool_reflection")) return "Reflecting on results";
            if (std.mem.eql(u8, stage, "compose_final_reply")) return "Preparing reply";
            if (std.mem.eql(u8, stage, "llm_first_token")) return "Model responding";
            if (std.mem.eql(u8, stage, "llm_first_token_upper_bound")) return "Waiting for model";
            if (std.mem.eql(u8, stage, "post_reply_compaction")) return "Compacting context";
            return null;
        }

        fn turnStageToFrameType(stage: []const u8) NarrationFrame.FrameType {
            // NOTE: FrameType reference here may need adjustment based on how
            // the type is ultimately accessed — use the module-level or struct-level path
            if (std.mem.eql(u8, stage, "dispatch_tools")) return .tool_start;
            if (std.mem.eql(u8, stage, "llm_first_token_upper_bound")) return .waiting;
            return .thinking;
        }
    };
    ```
