pub const OverallStatus = enum {
    healthy,

    pub fn fromChannels(entries: []const ChannelHealthEntry) OverallStatus {
        var worst: OverallStatus = .healthy;
        for (entries) |e| {
            switch (e.state) {
                .gave_up => return .unhealthy,
                .restarting => worst = .degraded,
                else => {},
            }
        }
        return worst;

    pub fn toSlice(self: OverallStatus) []const u8 {

**Score calculation** (Claude's discretion per CONTEXT.md):
```zig
pub fn calculateScore(results: []const SecurityCheckResult) u8 {
    var score: i16 = 100;
    for (results) |r| {
        if (r.severity != .ok) {
            score += @as(i16, r.score_impact);
        }
    }
    return @intCast(@max(0, @min(100, score)));
}

pub fn calculateGrade(score: u8) []const u8 {
    if (score >= 90) return "A";
    if (score >= 80) return "B";
    if (score >= 70) return "C";
    if (score >= 60) return "D";
    return "F";
}
```
