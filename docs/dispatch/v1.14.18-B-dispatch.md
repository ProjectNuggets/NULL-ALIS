  pub const PromotedSlot = struct {
      original_slot_id: i32,
      durable_key: []const u8,
      promoted_at_unix: i64,
  };
  
  /// Promote high-importance non-identity slots to durable_facts at session-end.
  /// Slots with composite_priority >= threshold + slot_type ∈ {active_goal, decision}
  /// become durable_facts with attribute=transient_goal.
  ///
  /// Called from session-end checkpoint path (commands.zig).
  /// At session-start, the memory_loader path already pulls durable_facts —
  /// so transient_goal facts re-enter the prompt context naturally.
  pub fn promoteWMToDurableAtSessionEnd(
      allocator: std.mem.Allocator,
      state_mgr: *zaki_state.Manager,
      mem: ?Memory,
      user_id: i64,
      session_id: []const u8,
  ) !u32 {  // returns count of promoted slots
      // 1. Read all WM slots for this session via listWorkingMemorySlots
      // 2. Filter: slot_type in {active_goal, decision} AND
      //            compositePriority(slot, now) >= 0.5
      // 3. For each: write to durable_facts via mem.store with:
      //    - key = "transient_goal/<session_id>/<slot_id>"
      //    - content = slot.content
      //    - attribute = "transient_goal"
      //    - predicate = slot.slot_type (e.g., "active_goal")
      // 4. Return count of writes
  }

Wire from `src/agent/commands.zig` session-end checkpoint path (alongside the
