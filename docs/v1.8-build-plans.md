   pub const ExtractionResult = struct {
       memories: []ExtractedMemory,
       
       pub fn deinit(self: *@This(), allocator: std.mem.Allocator) void { ... }
   };
   
   pub fn extract(allocator: std.mem.Allocator, user_message: []const u8) !ExtractionResult {
       var out: std.ArrayListUnmanaged(ExtractedMemory) = .empty;
       errdefer out.deinit(allocator);
       
       try extractIdentity(allocator, user_message, &out);     // "I am X", "my name is X"
       try extractRole(allocator, user_message, &out);         // "I work at X", "I work as X"
       try extractPreference(allocator, user_message, &out);   // "I prefer X over Y", "I like X"

       try extractEntities(allocator, user_message, &out);     // capitalized phrases ≥2 chars

**Goal:** single living "identity profile" memory block, refreshed on durable_fact writes; loaded into warm context every turn.

### Decisions

| # | Question | DECISION | Rationale |
|---|---|---|---|
| 9-A | Storage shape | Structured JSON memory block (per PL-6) | Diffable, atomically editable |
| 9-B | Key | `durable_fact/identity/profile` (single canonical key) | Matches existing namespace |
| 9-C | Aggregation source | `durable_fact/identity/*` + extracted triples WHERE subject="<self>" | Exhaustive identity coverage |
| 9-D | Refresh trigger | (a) any write to a key matching `durable_fact/identity/*` (b) session boundary | Cheap rebuild; bounded |
| 9-E | Warm-context injection | New `<active_identity>` block in memory_loader, alongside existing `<active_communities>` | Matches V1.7-ship S2c idiom |
| 9-F | Module location | `src/memory/lifecycle/identity_profile.zig` | Matches summarizer.zig sibling |

### Reuse Audit

- `commands.zig:1300` durable_fact write path (already calls syncVectorAfterStore)
- `memory_loader.zig::buildActiveCommunitiesBlock` (V1.7-ship) — exact same pattern, copy structure

### Implementation steps

1. Create `src/memory/lifecycle/identity_profile.zig`:
   ```zig
   pub const IdentityProfile = struct {
       name: ?[]const u8,
       roles: [][]const u8,
       preferences: [][]const u8,
       current_focus: [][]const u8,
       stakeholders: [][]const u8,
       
       pub fn fromState(allocator: std.mem.Allocator, state_mgr: *Manager, user_id: i64) !IdentityProfile { ... }
       pub fn toJson(self: IdentityProfile, allocator: std.mem.Allocator) ![]u8 { ... }
       pub fn render(self: IdentityProfile, allocator: std.mem.Allocator) ![]u8 { ... } // human-readable
   };
   
   pub fn rebuild(allocator: std.mem.Allocator, state_mgr: *Manager, user_id: i64) !void {
       const profile = try IdentityProfile.fromState(allocator, state_mgr, user_id);
       defer profile.deinit(allocator);
       const json = try profile.toJson(allocator);
       defer allocator.free(json);
       try state_mgr.upsertMemory(user_id, "durable_fact/identity/profile", json, .core, ...);
   }
   ```
2. Trigger rebuild at:
   - `commands.zig` durable_fact write path: after every write where `key starts with "durable_fact/identity/"`
   - `compaction.zig` session boundary path
3. Add `buildActiveIdentityBlock` to memory_loader.zig (mirror buildActiveCommunitiesBlock):
   ```zig
   pub fn buildActiveIdentityBlock(allocator: std.mem.Allocator, mem_rt: *MemoryRuntime, user_id: i64) ![]u8 {
       const profile_entry = try mem_rt.get(allocator, "durable_fact/identity/profile") orelse return "";
       defer profile_entry.deinit(allocator);
       return std.fmt.allocPrint(allocator, "<active_identity>\n{s}\n</active_identity>\n", .{profile_entry.content});
   }
   ```
4. Inject into warm context in turn-loop preflight; place near `<active_communities>` block
